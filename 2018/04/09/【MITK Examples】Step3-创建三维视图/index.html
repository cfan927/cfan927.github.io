【MITK Examples】Step3-创建三维视图

发表于 2018-04-09 | 分类于 mitk

原文参考：http://docs.mitk.org/nightly/Step03Page.html

1.设置图像文件、表面文件路径

和Step2相似，在Step3我们完成多个数据集的加载（多种图像、曲面和其它格式）。区别在于这次我们通过一个三维视图来显示。 把mapper-slot设置到Standard3D，可以实现使用窗口类QmitkRenderWindow来显示三维视图。因为体绘制是一个相当慢的处理过程 ，因此默认图像不在三维视图显示。 - STep3a为表面网格的默认条件工作。要使用Step3来运行这个示例，将参数传递为lungs.vtk（同Step1）。 - Step3b用于表面网格的体积绘制。要使用Step3运行这个示例，将参数传递为Pic3D.nrrd lungs.vtk（同Step2）。 - 设置完毕后，把MitkStep3设置为启动项目。

2.运行程序

2.1 代码结构

Part I :创建一个数据存储； Part II :从文件中读取数据； Part III:将数据放入数据存储中； Part IV:（Step 3a）通过dynamic_casting包含在节点中的数据来检查数据是否为图像； Part V :创建窗口并将数据存储传递给窗口，（Step 3b）以三维视图的方式显示； Part VI :Qt特定初始化。

2.2 程序代码

#include "QmitkRegisterClasses.h"
#include "QmitkRenderWindow.h"

#include <mitkIOUtil.h>
#include <mitkProperties.h>
#include <mitkRenderingManager.h>
#include <mitkStandaloneDataStorage.h>
#include <mitkTransferFunction.h>
#include <mitkTransferFunctionProperty.h>

#include <itksys/SystemTools.hxx>

#include <QApplication>

//## 转为三维视图显示
//## 就像在Step2中一样，加载一个或多个数据（如图像文件、表面文件格式），
//## 不过在这一节我们以三维视图的方式来显示。
//## 把mapper-slot设置到Standard3D，实现使用窗口类QmitkRenderWindow来显示三维视图。
//## 由于体渲染是一个相当缓慢的过程, 图像在默认情况下是不以三维视图的方式来显示的。
//## 在Step3示例中，我们想要进行体渲染，所以把布尔型的"volumerendering"切换为"true"。

int main(int argc, char *argv[])
{
  QApplication qtapplication(argc, argv);
  if (argc < 2)
  {
    fprintf(
      stderr, "Usage:   %s [filename1] [filename2] ...\n\n", itksys::SystemTools::GetFilenameName(argv[0]).c_str());
    return 1;
  }

  // 注册Qmitk相关的全局实例
  QmitkRegisterClasses();

  //*************************************************************************
  // Part I: 基本初始化
  //*************************************************************************

  // 创建一个数据存储
  mitk::StandaloneDataStorage::Pointer ds = mitk::StandaloneDataStorage::New();

  //*************************************************************************
  // Part II: 从文件中读取数据
  //*************************************************************************
  int i;
  for (i = 1; i < argc; ++i)
  {
    // For testing
    if (strcmp(argv[i], "-testing") == 0)
      continue;

    //*********************************************************************
    // Part III: 将数据放到数据存储中
    //*********************************************************************
    // 加载数据节点 (如图像文件格式，表面文件格式等)
    mitk::StandaloneDataStorage::SetOfObjects::Pointer dataNodes = mitk::IOUtil::Load(argv[i], *ds);

    if (dataNodes->empty())
    {
      fprintf(stderr, "Could not open file %s \n\n", argv[i]);
      exit(2);
    }
    mitk::DataNode::Pointer node = dataNodes->at(0);

    // *********************************************************
    // *********************新部分1开始(Step 3a) *****
    // *********************************************************

    //*********************************************************************
    // Part IV: 我们想要对所有的图像进行体渲染
    //*********************************************************************

    //通过dynamic_casting包含在节点中的数据来检查数据是否为图像。 警告：dynamic_cast的速度很慢，不要经常使用！
    mitk::Image::Pointer image = dynamic_cast<mitk::Image *>(node->GetData());
    if (image.IsNotNull())
    {
      // 将属性"volumerendering" 设为布尔值 "true"
      node->SetProperty("volumerendering", mitk::BoolProperty::New(true));

      // 创建传递函数以将光学属性（颜色和不透明度）分配给数据的灰度值
      mitk::TransferFunction::Pointer tf = mitk::TransferFunction::New();
      tf->InitializeByMitkImage(image);

      // 设置颜色传递函数 AddRGBPoint(double x, double r, double g, double b)
      tf->GetColorTransferFunction()->AddRGBPoint(tf->GetColorTransferFunction()->GetRange()[0], 1.0, 0.0, 0.0);
      tf->GetColorTransferFunction()->AddRGBPoint(tf->GetColorTransferFunction()->GetRange()[1], 1.0, 1.0, 0.0);

      // 设置分段不透明度传输函数 AddPoint(double x, double y)
      tf->GetScalarOpacityFunction()->AddPoint(0, 0);
      tf->GetScalarOpacityFunction()->AddPoint(tf->GetColorTransferFunction()->GetRange()[1], 1);

      node->SetProperty("TransferFunction", mitk::TransferFunctionProperty::New(tf.GetPointer()));
    }

    // *********************************************************
    // ******************* 新部分1结束 (Step 3a) *********
    // *********************************************************
  }

  //*************************************************************************
  // Part V: 创建一个渲染窗口并将数据存储传递给它
  //*************************************************************************

  // 创建一个渲染窗口
  QmitkRenderWindow renderWindow;

  // 告诉渲染窗口要渲染哪一个（哪一部分）的数据存储
  renderWindow.GetRenderer()->SetDataStorage(ds);

  // *********************************************************
  // ****************** 新部分2开始 (Step 3b) ********
  // *********************************************************
  // 设为三维视图
  renderWindow.GetRenderer()->SetMapperID(mitk::BaseRenderer::Standard3D);

  // 重新定位相机以包含所有可见的演员
  renderWindow.GetRenderer()->GetVtkRenderer()->ResetCamera();

  // *********************************************************
  // ******************* 新部分2结束(Step 3b) *********
  // *********************************************************

  //*************************************************************************
  // Part VI: Qt特定初始化
  //*************************************************************************
  renderWindow.show();
  renderWindow.resize(256, 256);

  mitk::RenderingManager::GetInstance()->RequestUpdateAll();

// 测试
#include "QtTesting.h"
  if (strcmp(argv[argc - 1], "-testing") != 0)
    return qtapplication.exec();
  else
    return QtTesting();
}
